# Step 1: Login to Docker registry
docker login docker-hosted.sebaoffice.ir

# Step 2: Build the Docker image
docker build -t docker-hosted.sebaoffice.ir/kss-mcp-template:latest .

# Alternative: Build and push in one command (if using buildx)
docker buildx build --push -t docker-hosted.sebaoffice.ir/kss-mcp-template:latest .

# Step 3: Push the image to registry
docker push docker-hosted.sebaoffice.ir/kss-mcp-template:latest

# Step 4: Create namespace (if it doesn't exist)
kubectl create namespace kss-mcp

# Step 5: Deploy/restart the application
kubectl rollout restart deployment/kss-mcp-template -n kss-mcp
